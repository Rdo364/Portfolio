Saudação com Nome

Um programa simples em Node.js que pede o nome do usuário e exibe uma mensagem de boas-vindas.

const readline = require('readline-sync');

function saudar (nome) {
  console.log(`Ola ${nome}, Seja bem vindo`);
}
let nome = readline.question("digite seu nome: ");
saudar(nome);

Cálculo de Média

Programa em Node.js que calcula a média de três notas e informa se o aluno foi aprovado ou reprovado.

const readline = require('readline-sync');

function media(n1, n2, n3) {
  let resultado = (n1 + n2 + n3) / 3;
  console.log(`media: ${resultado}`);

  if (resultado >= 6) {
    console.log("Aprovado");
  } else {
    console.log("Reprovado");
  }
}

let n1 = (readline.question("Diga a primeira nota "));
let n2 = (readline.question("Diga a segunda nota "));
let n3 = (readline.question("Diga a terceira nota "));

media(n1, n2, n3);
